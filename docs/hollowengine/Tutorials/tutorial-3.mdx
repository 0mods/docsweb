# Туториал №3 | Система диалога

:::note "Создадим простенький диалог в спец. окне и последствиями"

```kts
dialogue{
    npc say{"Привет"}
	choiсe{
		"Приветик!"{
			team.sendAsPlayer{"Давно не виделись!"}
			npc say{"Как дела хоть?"}
			choise{
				"Всё просто супер"{
					npc say{"Рад слышать это)"}
				}
				"Убейте меня."{
					npc say{"Ну.. сам попросил."}
					wait{1.sec}
					npc say{"Потом не ной!"}
				}
			}
		}
		"Эм.." {
			npc playLooped{"face-what"}
			npc say{"Что-то случилось"}
			choise{
				"А? да неет.."{
					team.sendAsPlayer{"Да не, всё ок. Просто голова чёт заболела"}
				}
				"Ты кто?"{
					npc say{"я в тотальном шоке."}
				}
			}
		}
	}
}
```
:::
